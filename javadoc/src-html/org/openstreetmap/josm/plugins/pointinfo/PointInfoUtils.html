002package org.openstreetmap.josm.plugins.pointinfo;
003
004import java.text.DecimalFormat;
005import java.text.DecimalFormatSymbols;
006
007import javax.swing.JOptionPane;
008
009import org.openstreetmap.josm.gui.Notification;
010
011public final class PointInfoUtils {
012
013    private PointInfoUtils() {
014        // Hide default constructor for utilities classes
015    }
016
017    /**
018     * Show notification.
019     * @param message Message to shown.
020     * @param type Type if message (info, warning, error, plain).
021     */
022    public static void showNotification(String message, String type) {
023        Notification note = new Notification(message);
024
025        if ("info".equals(type))
026            note.setIcon(JOptionPane.INFORMATION_MESSAGE);
027        else if ("warning".equals(type))
028            note.setIcon(JOptionPane.WARNING_MESSAGE);
029        else if ("error".equals(type))
030            note.setIcon(JOptionPane.ERROR_MESSAGE);
031        else
032            note.setIcon(JOptionPane.PLAIN_MESSAGE);
033
034        note.setDuration(Notification.TIME_SHORT);
035        note.show();
036    }
037
038    /**
039     * Return text representation of coordinates.
040     * @param lat the lat part of coordinates
041     * @param lon the lon part of coordinates
042     * @return String coordinatesText
043     */
044    public static String formatCoordinates(double lat, double lon) {
045
046        DecimalFormatSymbols symbols = new DecimalFormatSymbols();
047        symbols.setDecimalSeparator('.');
048        symbols.setGroupingSeparator(' ');
049        DecimalFormat df = new DecimalFormat("#.00000", symbols);
050
051        return "(" + df.format(lat) + ", " + df.format(lon) + ")";
052    }
053}
