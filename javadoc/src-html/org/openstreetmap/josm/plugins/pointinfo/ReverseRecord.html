002package org.openstreetmap.josm.plugins.pointinfo;
003
004import javax.json.JsonObject;
005
006/**
007 * Class to contain Nominatim Reverse Geocoding data
008 * @author Javier SÃ¡nchez Portero
009 */
010class ReverseRecord {
011
012    private String countryCode;
013    private String country;
014    private String state;
015    private String stateDistrict;
016    private String county;
017    private String city;
018    private String town;
019    private String village;
020    private String cityDistrict;
021    private String suburb;
022
023    /**
024     * Default constructor
025     *
026     */
027    ReverseRecord() {
028        init();
029    }
030
031    /**
032     * Constructor from JSON
033     * @param obj the Json object
034     *
035     */
036    ReverseRecord(JsonObject obj) {
037        init();
038        JsonObject address = obj.getJsonObject("address");
039        if (address != null) {
040            countryCode = address.getString("country_code", null);
041            country = address.getString("country", null);
042            state = address.getString("state", null);
043            stateDistrict = address.getString("state_district", null);
044            county = address.getString("county", null);
045            city = address.getString("city", null);
046            town = address.getString("town", null);
047            village = address.getString("village", null);
048            cityDistrict = address.getString("city_district", null);
049            suburb = address.getString("suburb", null);
050        }
051    }
052
053    /**
054     * Initialization
055     *
056     */
057    private void init() {
058        countryCode = null;
059        country = null;
060        state = null;
061        stateDistrict = null;
062        county = null;
063        city = null;
064        town = null;
065        village = null;
066        cityDistrict = null;
067        suburb = null;
068    }
069
070    /**
071     * Returns true if area is equals to any address value
072     * @param area area to be checked
073     * @return match area matched
074     */
075    public Boolean matchAnyArea(String area) {
076        if (area.equals(countryCode)) return true;
077        if (area.equals(country)) return true;
078        if (area.equals(state)) return true;
079        if (area.equals(stateDistrict)) return true;
080        if (area.equals(county)) return true;
081        if (area.equals(city)) return true;
082        if (area.equals(town)) return true;
083        if (area.equals(village)) return true;
084        if (area.equals(cityDistrict)) return true;
085        if (area.equals(suburb)) return true;
086        return false;
087    }
088}
